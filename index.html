export class Async {
    async get(url: string) {
        const response = await fetch(url)
        const resData = await response.json()
        return resData
      }
      async post(url: string, data: {name: string, color: string}) {
        const response = await fetch(url, {
          method: 'POST',
          headers: {
            'Content-type': 'application/json'
          },
          body: JSON.stringify(data)
        });
        const resData = await response.json()
        return resData;
      }
